        </div>` : ''}
        <div class="size-row">
          ${p.sizes.map(s => `<button class="sbtn ${ss===s?'sel':''}" onclick="pickSize(${p.id},'${s}')">${s}</button>`).join('')}
        </div>
        <button class="add-btn" onclick="addToCart(${p.id})">${t.addBtn}</button>
      </div>
    </div>`;
  }).join('');
}

function pickColor(pid, ci) { selColors[pid]=ci; renderShop(); }
function pickSize(pid, s) { selSizes[pid]=s; renderShop(); }

function addToCart(pid) {
  const l = currentLang;
  const t = T[l];
  const size = selSizes[pid];
  if (!size) { showToast(t.sizeMsg); return; }
  const p = PRODUCTS_DATA.find(x=>x.id===pid);
  const pInfo = t.products[pid-1];
  const ci = selColors[pid]??0;
  const color = p.colorN[l][ci];
  const price = l==='pl'?p.pln:p.uah;
  const sym = l==='pl'?'zł':'₴';
  cart.push({pid, name:pInfo.name, size, color, price, sym, img:p.img});
  document.getElementById('cart-count').textContent = cart.length;
  renderCart();
  showToast(t.addedMsg);
}

/* ── CART ── */
function renderCart() {
  const l = currentLang;
  const t = T[l];
  const body = document.getElementById('cart-body');
  const foot = document.getElementById('cart-foot');
  if (!cart.length) {
    body.innerHTML = `<div class="cart-empty-state">
      <span class="cart-empty-icon">✦</span>
      <span class="cart-empty-text">${t.emptyCart}</span>
    </div>`;
    foot.style.display='none'; return;
  }
  body.innerHTML = cart.map((item,i) => `
    <div class="cart-item">
      <img class="cart-item-img" src="${item.img}" alt="${item.name}">
      <div class="cart-item-info">
        <div class="cart-item-name">${item.name}</div>
        <div class="cart-item-meta">${item.size} · ${item.color}</div>
        <div class="cart-item-price">${item.price.toLocaleString()} ${item.sym}</div>
      </div>
      <button class="cart-rm" onclick="removeItem(${i})">×</button>
    </div>`).join('');
  const total = cart.reduce((s,i)=>s+i.price,0);
  const sym = cart[0]?.sym||'₴';
  document.getElementById('cart-total').textContent = total.toLocaleString()+' '+sym;
  foot.style.display='block';
}

function removeItem(i) {
  cart.splice(i,1);
  document.getElementById('cart-count').textContent = cart.length;
  renderCart();
}

function openCart() {
  document.getElementById('cart-drawer').classList.add('open');
  document.getElementById('overlay').classList.add('open');
}
function closeCart() {
  document.getElementById('cart-drawer').classList.remove('open');
  document.getElementById('overlay').classList.remove('open');
}

function showToast(msg) {
  const el = document.getElementById('toast');
  el.textContent = msg;
  el.classList.add('show');
  setTimeout(() => el.classList.remove('show'), 2200);
}
